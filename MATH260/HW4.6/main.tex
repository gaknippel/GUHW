\section*{problem}

find particular solution to:
\[
    y'' + y = \tan t + \sin t + 1
\]

\section*{solve the homogeneous equation}

solve $y'' + y = 0$. characteristic equation is $r^2 + 1 = 0$, giving $r = \pm i$. thus the homogeneous solution is:
\[
    y_h = c_1 \cos t + c_2 \sin t
\]

our two fundamental solutions are $y_1 = \cos t$ and $y_2 = \sin t$.

\section*{compute the wronskian}

\[
    W = \begin{vmatrix} y_1 & y_2 \\ y_1' & y_2' \end{vmatrix}
      = \begin{vmatrix} \cos t & \sin t \\ -\sin t & \cos t \end{vmatrix}
      = \cos^2 t + \sin^2 t = 1
\]

\section*{set up the variation of parameters}

we hgave a particular solution of form $y_p = u_1(t)\cos t + u_2(t)\sin t$, where $g(t) = \tan t + \sin t + 1$. the formulas give:

\[
    u_1' = \frac{-y_2 \cdot g(t)}{W} = -\sin t\,(\tan t + \sin t + 1)
\]
\[
    u_2' = \frac{y_1 \cdot g(t)}{W} = \cos t\,(\tan t + \sin t + 1)
\]

\section*{integrate $u_1$}

\[
    u_1' = -\sin t \cdot \tan t - \sin^2 t - \sin t
           = -\frac{\sin^2 t}{\cos t} - \frac{1 - \cos 2t}{2} - \sin t
\]

integrating:
\begin{align*}
    \int -\frac{\sin^2 t}{\cos t}\,dt
        &= \int \frac{\cos^2 t - 1}{\cos t}\,dt
         = \int \bigl(\cos t - \sec t\bigr)\,dt
         = \sin t - \ln|\sec t + \tan t| \\[6pt]
    \int -\frac{1 - \cos 2t}{2}\,dt
        &= -\frac{t}{2} + \frac{\sin 2t}{4} \\[6pt]
    \int -\sin t\,dt
        &= \cos t
\end{align*}

so:
\[
    u_1 = \sin t - \ln|\sec t + \tan t| - \frac{t}{2} + \frac{\sin 2t}{4} + \cos t
\]

\section*{integrate $u_2$}

\[
    u_2' = \cos t \cdot \tan t + \cos t \sin t + \cos t
           = \sin t + \frac{\sin 2t}{2} + \cos t
\]

integrating:
\begin{align*}
    \int \sin t\,dt &= -\cos t \\[6pt]
    \int \frac{\sin 2t}{2}\,dt &= -\frac{\cos 2t}{4} \\[6pt]
    \int \cos t\,dt &= \sin t
\end{align*}

so:
\[
    u_2 = -\cos t - \frac{\cos 2t}{4} + \sin t
\]

\section*{put it all together $y_p = u_1 \cos t + u_2 \sin t$}

substituting and expanding (using $\sin 2t = 2\sin t \cos t$ and $\cos 2t = 1 - 2\sin^2 t$):

\begin{align*}
    y_p &= \cos t \!\left[\sin t - \ln|\sec t + \tan t| - \tfrac{t}{2} + \tfrac{\sin 2t}{4} + \cos t\right] \\
        &\quad + \sin t \!\left[-\cos t - \tfrac{\cos 2t}{4} + \sin t\right]
\end{align*}

after expanding and canceling terms from $y_h$

\[
    \boxed{y_p = 1 - \frac{t}{2}\cos t - \cos t \ln|\sec t + \tan t|}
\]

\section*{general solution}

\[
    y = c_1 \cos t + c_2 \sin t + 1 - \frac{t}{2}\cos t - \cos t \ln|\sec t + \tan t|
\]
